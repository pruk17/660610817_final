#!/usr/bin/env python3
"""
UDP_sender.py (node1_udp_gateway_agv) ? Run on AGV (ROS_DOMAIN_ID=0)

Acts as a two-way UDP bridge between AGV and Laptop:
  - Receives velocity commands from Laptop via UDP port 15000
    and publishes them to /cmd_vel (ROS2 -> ROS1 bridge -> motor)
  - Subscribes to /scan_distance (String) and forwards each message
    to the Laptop via UDP port 15001

Watchdog: if no UDP command is received within cmd_timeout seconds,
a zero Twist is published to stop the robot safely.

Run:
  ROS_DOMAIN_ID=0 python3 UDP_sender.py \
    --ros-args -p laptop_ip:=10.244.171.183
"""
import socket
import threading
import time
import rclpy
from rclpy.node import Node
from rclpy.qos import QoSProfile, ReliabilityPolicy
from geometry_msgs.msg import Twist
from std_msgs.msg import String


class UdpGatewayAgv(Node):

    def __init__(self):
        super().__init__('udp_gateway_agv')

        # Parameters
        self.declare_parameter('laptop_ip',   '10.244.171.183')
        self.declare_parameter('cmd_port',    15000)
        self.declare_parameter('data_port',   15001)
        self.declare_parameter('cmd_timeout', 0.5)

        self.laptop_ip   = self.get_parameter('laptop_ip').value
        self.cmd_port    = self.get_parameter('cmd_port').value
        self.data_port   = self.get_parameter('data_port').value
        self.cmd_timeout = self.get_parameter('cmd_timeout').value

        qos = QoSProfile(depth=10, reliability=ReliabilityPolicy.BEST_EFFORT)

        # Publisher: received UDP cmd -> /cmd_vel -> ros1_bridge -> motor
        self.cmd_pub = self.create_publisher(Twist, '/cmd_vel', qos)

        # UDP socket: receive velocity commands from Laptop
        self.sock_recv = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        self.sock_recv.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        self.sock_recv.bind(('0.0.0.0', self.cmd_port))
        self.last_rx = time.time()

        # Start UDP receive loop in background thread
        threading.Thread(target=self._cmd_loop, daemon=True).start()

        # Watchdog timer: publish zero if no command received recently
        self.create_timer(0.1, self._watchdog)

        # UDP socket: send /scan_distance data to Laptop
        self.sock_send = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        self.create_subscription(String, '/scan_distance', self._dist_cb, qos)

        self.get_logger().info(f'[AGV | DOMAIN 0] UDP:{self.cmd_port} -> /cmd_vel')
        self.get_logger().info(
            f'[AGV | DOMAIN 0] /scan_distance -> UDP {self.laptop_ip}:{self.data_port}')

    def _cmd_loop(self):
        """Background thread: receive UDP packets and publish as Twist."""
        self.sock_recv.settimeout(1.0)
        while rclpy.ok():
            try:
                data, _ = self.sock_recv.recvfrom(2048)
                parts = data.decode().strip().split()
                if len(parts) == 3:
                    x, y, z = map(float, parts)
                    msg = Twist()
                    msg.linear.x  = x
                    msg.linear.y  = y
                    msg.angular.z = z
                    self.cmd_pub.publish(msg)
                    self.last_rx = time.time()
            except socket.timeout:
                pass
            except Exception as e:
                if rclpy.ok():
                    self.get_logger().warn(f'CMD recv error: {e}')

    def _watchdog(self):
        """Publish zero velocity if Laptop connection is lost."""
        if time.time() - self.last_rx > self.cmd_timeout:
            self.cmd_pub.publish(Twist())

    def _dist_cb(self, msg: String):
        """Forward /scan_distance string to Laptop over UDP."""
        self.sock_send.sendto(
            msg.data.encode(), (self.laptop_ip, self.data_port))


def main(args=None):
    rclpy.init(args=args)
    node = UdpGatewayAgv()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.cmd_pub.publish(Twist())   # stop robot on exit
        node.destroy_node()
        if rclpy.ok():
            rclpy.shutdown()


if __name__ == '__main__':
    main()
