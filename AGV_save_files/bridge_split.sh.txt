#!/bin/bash

SESSION=bridge_session
PYTHON_GALACTIC="PYTHONPATH=/opt/ros/galactic/lib/python3.8/site-packages"

tmux kill-session -t $SESSION 2>/dev/null
tmux new-session -d -s $SESSION

# -------------------------
# ros1_bridge ( ROS1 + ROS2)
# -------------------------
tmux send-keys -t $SESSION "source /opt/ros/noetic/setup.bash" C-m
tmux send-keys -t $SESSION "source /opt/ros/galactic/setup.bash" C-m
tmux send-keys -t $SESSION "export ROS_DOMAIN_ID=0" C-m
tmux send-keys -t $SESSION "ros2 run ros1_bridge dynamic_bridge --bridge-all-topics" C-m

# -------------------------
# scan filter (ROS2 only)
# -------------------------
tmux split-window -h -t $SESSION
tmux send-keys -t $SESSION "unset ROS_DISTRO ROS_VERSION ROS_PACKAGE_PATH" C-m
tmux send-keys -t $SESSION "source /opt/ros/galactic/setup.bash" C-m
tmux send-keys -t $SESSION "export ROS_DOMAIN_ID=0" C-m
tmux send-keys -t $SESSION "$PYTHON_GALACTIC python3 ~/660610817_ws/scan_filter_node.py" C-m

# -------------------------
# UDP sender (ROS2 only)
# -------------------------
tmux split-window -h -t $SESSION
tmux send-keys -t $SESSION "unset ROS_DISTRO ROS_VERSION ROS_PACKAGE_PATH" C-m
tmux send-keys -t $SESSION "source /opt/ros/galactic/setup.bash" C-m
tmux send-keys -t $SESSION "export ROS_DOMAIN_ID=0" C-m
tmux send-keys -t $SESSION "$PYTHON_GALACTIC python3 ~/660610817_ws/UDP_sender.py" C-m

tmux select-layout -t $SESSION even-horizontal

tmux attach -t $SESSION
